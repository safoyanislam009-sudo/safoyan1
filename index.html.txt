    <div class="container">
        <div class="side">
            <h2>Your Offer</h2>
            <select id="selectA" onchange="addFruit('A')">
                <option value="">+ Add Fruit</option>
            </select>
            <div id="listA" class="list"></div>
            <div class="total">Total: <span id="totalA">0</span></div>
        </div>

        <div class="side">
            <h2>Their Offer</h2>
            <select id="selectB" onchange="addFruit('B')">
                <option value="">+ Add Fruit</option>
            </select>
            <div id="listB" class="list"></div>
            <div class="total">Total: <span id="totalB">0</span></div>
        </div>

        <div class="result-bar" id="status">Add fruits to see trade value</div>
    </div>

    <script>
        // MARCH 2026 DATABASE
        const fruitData = {
            "Kitsune": 115000000, "Dragon": 90000000, "Leopard": 40000000, "Dough": 25000000,
            "T-Rex": 20000000, "Mammoth": 12500000, "Spirit": 9000000, "Venom": 9000000,
            "Control": 8000000, "Buddha": 7000000, "Portal": 6000000, "Blizzard": 5000000,
            "Rumble": 5000000, "Shadow": 3000000, "Sound": 3000000, "Phoenix": 2000000,
            "Magma": 1500000, "Light": 1000000, "Quake": 1000000, "Dark": 500000
        };

        let sideA = [];
        let sideB = [];

        function init() {
            const sels = [document.getElementById('selectA'), document.getElementById('selectB')];
            Object.keys(fruitData).sort().forEach(name => {
                sels.forEach(s => {
                    let opt = document.createElement('option');
                    opt.value = name;
                    opt.innerText = `${name} (${fruitData[name].toLocaleString()})`;
                    s.appendChild(opt);
                });
            });
        }

        function addFruit(side) {
            let sel = document.getElementById('select' + side);
            if (!sel.value) return;
            if (side === 'A') sideA.push(sel.value);
            else sideB.push(sel.value);
            sel.value = "";
            render();
        }

        function removeFruit(side, index) {
            if (side === 'A') sideA.splice(index, 1);
            else sideB.splice(index, 1);
            render();
        }

        function render() {
            let valA = 0, valB = 0;
            document.getElementById('listA').innerHTML = sideA.map((f, i) => {
                valA += fruitData[f];
                return `<div class="fruit-item"><span>${f}</span><span class="remove-btn" onclick="removeFruit('A', ${i})">X</span></div>`;
            }).join('');
            
            document.getElementById('listB').innerHTML = sideB.map((f, i) => {
                valB += fruitData[f];
                return `<div class="fruit-item"><span>${f}</span><span class="remove-btn" onclick="removeFruit('B', ${i})">X</span></div>`;
            }).join('');

            document.getElementById('totalA').innerText = valA.toLocaleString();
            document.getElementById('totalB').innerText = valB.toLocaleString();

            updateStatus(valA, valB);
        }

        function updateStatus(a, b) {
            const status = document.getElementById('status');
            if (a === 0 || b === 0) { status.innerText = "Add fruits to both sides"; return; }
            
            let diff = a - b;
            let percent = (Math.abs(diff) / a) * 100;

            if (percent < 10) { status.className = "result-bar fair"; status.innerText = "Fair Trade"; }
            else if (a < b) { status.className = "result-bar win"; status.innerText = "Big Win! (+" + Math.round(percent) + "%)"; }
            else { status.className = "result-bar loss"; status.innerText = "L / Loss (" + Math.round(percent) + "%)"; }
        }

        init();
    </script>
